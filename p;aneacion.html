# Script para generar una planeación semanal CONAFE def generar_planeacion(): print("=== Planeación Semanal CONAFE ===\n") maestro = input("Nombre del maestro/a: ") grupo = input("Grupo o multigrado: ") semana = input("Semana (ej. 1 al 5 de marzo): ") asignatura = input("Asignatura: ") tema = input("Tema central: ") proposito = input("Propósito de la semana: ") materiales = input("Materiales didácticos: ") actividades = [] dias = ["Lunes", "Martes", "Miércoles", "Jueves", "Viernes"] for dia in dias: act = input(f"Actividad principal para {dia}: ") actividades.append(act) print("\n=== PLANEACIÓN SEMANAL ===") print(f"Maestro/a: {maestro}") print(f"Grupo: {grupo}") print(f"Semana: {semana}") print(f"Asignatura: {asignatura}") print(f"Tema: {tema}") print(f"Propósito: {proposito}") print(f"Materiales: {materiales}") print("\nActividades diarias:") for i, dia in enumerate(dias): print(f"{dia}: {actividades[i]}") print("\n¡Planeación generada exitosamente!") if __name__ == "__main__": generar_planeacion()
